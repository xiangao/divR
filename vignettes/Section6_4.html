Introduction

Instrumental variable estimation relies on the instrument \(Z\) being relevant to the treatment \(X\). In this section, we explore how divR performs when the instrument is weak.

Data Generating Process

We use a parameter \(\alpha\) to control the instrument strength: - \(X = \alpha(Z + H) + 0.5\epsilon_X\) - \(Y = \tanh(1.5X) - 2.5H + \epsilon_Y\)

When \(\alpha\) is small, the instrument is “weak”.

g_fct <- function(Z, H, eps_X, alpha) alpha * (Z + H) + 0.5 * eps_X
f_tanh <- function(X, H, eps_Y) tanh(1.5 * X) - 2.5 * H + eps_Y

simulate_data <- function(alpha, n = 2000) {
  H <- runif(n, -1, 1)
  Z <- runif(n, -3, 3)
  eps_X <- runif(n, -1, 1)
  eps_Y <- runif(n, -1, 1)
  X <- g_fct(Z, H, eps_X, alpha)
  Y <- f_tanh(X, H, eps_Y)
  list(Z=Z, X=X, Y=Y)
}

Comparison: Strong vs Weak Instrument

we fit the model for \(\alpha = 5\) (strong) and \(\alpha = 0.5\) (weak).

# Strong Instrument (alpha = 5)
d_strong <- simulate_data(alpha = 5)
mod_strong <- divR(Z = matrix(d_strong$Z, ncol=1), X = matrix(d_strong$X, ncol=1), Y = matrix(d_strong$Y, ncol=1), num_epochs = 1000, silent = TRUE)

# Weak Instrument (alpha = 0.5)
d_weak <- simulate_data(alpha = 0.5)
mod_weak <- divR(Z = matrix(d_weak$Z, ncol=1), X = matrix(d_weak$X, ncol=1), Y = matrix(d_weak$Y, ncol=1), num_epochs = 1000, silent = TRUE)

Results

X_grid <- matrix(seq(-2, 2, length.out = 100), ncol = 1)

# True mean
get_true_mean <- function(x_val, n_rep = 5000) {
  H_rep <- runif(n_rep, -1, 1)
  eps_Y_rep <- runif(n_rep, -1, 1)
  mean(f_tanh(rep(x_val, n_rep), H_rep, eps_Y_rep))
}
true_means <- sapply(X_grid, get_true_mean)

pred_strong <- predict(mod_strong, Xtest = X_grid, type = "mean")
pred_weak <- predict(mod_weak, Xtest = X_grid, type = "mean")

plot_df <- data.frame(
  X = as.vector(X_grid),
  Strong = as.vector(pred_strong),
  Weak = as.vector(pred_weak),
  True = true_means
)

ggplot(plot_df, aes(x = X)) +
  geom_line(aes(y = Strong, color = "Strong Instrument (alpha=5)"), size = 1) +
  geom_line(aes(y = Weak, color = "Weak Instrument (alpha=0.5)"), size = 1) +
  geom_line(aes(y = True, color = "True Causal"), linetype = "dashed", size = 1) +
  labs(title = "Effect of Instrument Strength on divR Estimates",
       y = "E[Y | do(X)]", color = "Model") +
  theme_minimal()
