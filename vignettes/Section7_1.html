Introduction

This vignette replicates the analysis of Acemoglu et al. (2001) using the divR package. The study investigates the effect of institutional quality (average protection against expropriation risk) on economic performance (log GDP per capita in 1995), using log European settler mortality as an instrument.

df <- read.dta13(dta_file)
df <- df %>% filter(baseco == 1)

# Variables
X <- df$avexpr  # Institutional Quality
Z <- df$logem4 # Log Settler Mortality
Y <- df$logpgp95 # Log GDP

Fit divR Model

# We use a smaller number of epochs for the vignette to speed up rendering
model <- divR(Z = matrix(Z, ncol=1), 
              X = matrix(X, ncol=1), 
              Y = matrix(Y, ncol=1), 
              num_epochs = 5000, 
              silent = TRUE)

Comparison with OLS and 2SLS

avexpr_seq <- matrix(seq(min(X), max(X), length.out = 100), ncol = 1)

# divR predictions
mean_pred_div <- predict(model, Xtest = avexpr_seq, type = "mean", n_samples = 1000)
q_pred_div <- predict(model, Xtest = avexpr_seq, type = "quantile", 
                      probs = c(0.1, 0.9), n_samples = 1000)

# OLS for comparison
ols_mod <- lm(Y ~ X)
ols_pred <- predict(ols_mod, newdata = data.frame(X = as.vector(avexpr_seq)))

plot_df <- data.frame(
  avexpr = as.vector(avexpr_seq),
  DIV_mean = as.vector(mean_pred_div),
  DIV_lower = q_pred_div[, 1],
  DIV_upper = q_pred_div[, 2],
  OLS = ols_pred
)

ggplot() +
  geom_point(data = df, aes(x = avexpr, y = logpgp95), alpha = 0.5) +
  geom_line(data = plot_df, aes(x = avexpr, y = DIV_mean, color = "divR Mean"), size = 1.2) +
  geom_ribbon(data = plot_df, aes(x = avexpr, ymin = DIV_lower, ymax = DIV_upper), 
              fill = "green", alpha = 0.2) +
  geom_line(data = plot_df, aes(x = avexpr, y = OLS, color = "OLS"), linetype = "dashed") +
  labs(title = "Colonial Origins: divR vs OLS",
