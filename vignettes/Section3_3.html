Data Generating Process

In Section 3.3, the following softplus DGP is used: - \(g(Z, H, \epsilon_X) = \log(1 + \exp(Z + H + \epsilon_X))\) - \(f(X, H, \epsilon_Y) = 0.5 \cdot \log(1 + \exp(2X + 2H + 4\epsilon_Y))\)

g_softplus <- function(Z, H, eps_X) log(1 + exp(Z + H + eps_X))
f_softplus <- function(X, H, eps_Y) 0.5 * log(1 + exp(2 * X + 2 * H + 4 * eps_Y))

set.seed(1994)
n_train <- 2000
H <- rnorm(n_train)
Z <- runif(n_train, 0, 3)
eps_X <- rnorm(n_train)
eps_Y <- rnorm(n_train)

X <- g_softplus(Z, H, eps_X)
Y <- f_softplus(X, H, eps_Y)

df_train <- data.frame(Z=Z, X=X, Y=Y)

Fit divR Model

model <- divR(Z = matrix(Z, ncol=1), 
              X = matrix(X, ncol=1), 
              Y = matrix(Y, ncol=1), 
              num_epochs = 1000, 
              silent = TRUE)

Interventional Prediction

We compare the model’s interventional mean prediction \(E[Y | do(X)]\) with the true interventional mean.

X_grid <- matrix(seq(min(X), max(X), length.out = 100), ncol = 1)
Y_causal_pred <- predict(model, Xtest = X_grid, type = "mean", n_samples = 500)

# True interventional mean via simulation
get_true_mean <- function(x_val, n_rep = 5000) {
  H_rep <- rnorm(n_rep)
  eps_Y_rep <- rnorm(n_rep)
  Y_rep <- f_softplus(rep(x_val, n_rep), H_rep, eps_Y_rep)
  mean(Y_rep)
}

true_means <- sapply(X_grid, get_true_mean)

plot_df <- data.frame(
  X = as.vector(X_grid),
  Predicted = as.vector(Y_causal_pred),
  True = true_means
)

ggplot(plot_df, aes(x = X)) +
  geom_line(aes(y = Predicted, color = "divR Estimate"), size = 1) +
  geom_line(aes(y = True, color = "True Causal"), linetype = "dashed", size = 1) +
  labs(title = "Section 3.3 Softplus Example",
       y = "E[Y | do(X)]",
       color = "Legend") +
  theme_minimal()
